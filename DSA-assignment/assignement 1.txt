#include <stdio.h>
#include <string.h>

#define MAX 100

// Stack structure
char stack[MAX];
int top = -1;

// push function
void push(char c) {
    if (top < MAX - 1) {
        stack[++top] = c;
    }
}

// pop function
char pop() {
    if (top >= 0) {
        return stack[top--];
    }
    return '\0';
}

// check matching pair
int isMatching(char open, char close) {
    if (open == '(' && close == ')') return 1;
    if (open == '[' && close == ']') return 1;
    if (open == '{' && close == '}') return 1;
    return 0;
}

// function to check balance
int isBalanced(char expr[]) {
    top = -1;  // reset stack

    for (int i = 0; i < strlen(expr); i++) {
        char ch = expr[i];

        if (ch == '(' || ch == '[' || ch == '{') {
            push(ch);
        }
        else if (ch == ')' || ch == ']' || ch == '}') {
            if (top == -1) return 0;
            char open = pop();
            if (!isMatching(open, ch)) return 0;
        }
    }
    return top == -1;
}

int main() {

    char *tests[] = {
        "a + (b - c) * (d",
        "m + [a - b * (c + d * {m)]",
        "a + (b - c)"
    };

    for (int i = 0; i < 3; i++) {
        printf("Expression: %s\n", tests[i]);
        if (isBalanced(tests[i]))
            printf("Balanced\n\n");
        else
            printf("Unbalanced\n\n");
    }

    return 0;
}
